## Processing

for stem in \
  empyrean_dragon_basic empyrean_dragon_gold \
  aurora_leviathan_basic aurora_leviathan_gold \
  glacial_seraph_basic glacial_seraph_gold \
  seraph_owl_basic seraph_owl_gold \
  aurora_fox_basic aurora_fox_gold \
  frostlight_hare_basic frostlight_hare_gold \
  sunmane_lion_basic sunmane_lion_gold \
  solar_phoenix_basic solar_phoenix_gold \
  radiant_salamander_basic radiant_salamander_gold \
  emberling_cherub_basic emberling_cherub_gold; do
  bash scripts/decimate_mesh.sh "assets/source/pets/${stem}.obj" \
    --output "assets/exports/pets/${stem}" \
    --targets 5000,7500,10000
done
