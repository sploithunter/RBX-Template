## Processing

for stem in \
  solar_roc_basic solar_roc_gold \
  dawn_camel_basic dawn_camel_gold \
  gilded_sphinx_basic gilded_sphinx_gold \
  mirage_jackal_basic mirage_jackal_gold \
  sun_scarab_basic sun_scarab_gold; do
  bash scripts/decimate_mesh.sh "assets/source/pets/${stem}.obj" \
    --output "assets/exports/pets/${stem}" \
    --targets 5000,7500,10000
done
