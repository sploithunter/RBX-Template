## Processing

for stem in carrion_vulture_basic golden_scarab_basic sand_scorpion_basic dune_tortoise_basic sand_jackal_basic; do
  bash scripts/decimate_mesh.sh "assets/source/pets/${stem}.obj" \
    --output "assets/exports/pets/${stem}" \
    --targets 5000,7500,10000
done
