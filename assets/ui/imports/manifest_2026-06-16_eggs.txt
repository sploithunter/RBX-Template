# Egg 3D mesh import (2026-06-16)
# Pipeline: import zip -> assets/source/eggs/<egg_id>/ -> decimate_mesh.sh -> assets/exports/eggs/<egg_id>/

for egg_id in ash_egg blight_egg black_ice_egg infernal_egg gilded_egg bloom_egg aurora_egg solar_egg; do
  bash scripts/decimate_mesh.sh "assets/source/eggs/${egg_id}/${egg_id}.obj" \
    --output "assets/exports/eggs/${egg_id}" \
    --targets 5000,7500,10000
done
