## Processing

for stem in \
  bloomlamb_basic bloomlamb_gold \
  worldroot_ent_basic worldroot_ent_gold \
  verdant_sprite_basic verdant_sprite_gold \
  goldleaf_stag_basic goldleaf_stag_gold \
  halo_hare_basic halo_hare_gold; do
  bash scripts/decimate_mesh.sh "assets/source/pets/${stem}.obj" \
    --output "assets/exports/pets/${stem}" \
    --targets 5000,7500,10000
done
